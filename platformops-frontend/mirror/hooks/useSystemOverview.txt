import { useQuery } from "@tanstack/react-query";
import { fetchSystemOverview } from "../api/system.api";

export function useSystemOverview() {
  return useQuery({
    queryKey: ["system", "overview"],
    queryFn: fetchSystemOverview,
    refetchInterval: 5000,
  });
}
