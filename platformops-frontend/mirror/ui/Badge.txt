export function Badge({
  label,
  tone,
}: {
  label: string;
  tone: "ok" | "warn" | "crit";
}) {
  const map = {
    ok: "bg-green-900 text-green-300",
    warn: "bg-yellow-900 text-yellow-300",
    crit: "bg-red-900 text-red-300",
  };
  return (
    <span className={`rounded px-2 py-0.5 text-xs ${map[tone]}`}>
      {label}
    </span>
  );
}
